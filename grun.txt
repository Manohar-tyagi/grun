<script>
const audioCtx = new (window.AudioContext || window.webkitAudioContext)();
function playNeonSound(type) {
    if(audioCtx.state === 'suspended') audioCtx.resume();
    const osc = audioCtx.createOscillator();
    const gain = audioCtx.createGain();
    osc.connect(gain);
    
    const now = audioCtx.currentTime;
    
    if (type === 'jump') {
        gain.connect(audioCtx.destination);
        osc.type = 'sine';
        osc.frequency.setValueAtTime(300, now);
        osc.frequency.exponentialRampToValueAtTime(700, now + 0.15);
        gain.gain.setValueAtTime(0.3, now);
        gain.gain.exponentialRampToValueAtTime(0.01, now + 0.15);
        osc.start(now);
        osc.stop(now + 0.15);
    } 
    else if (type === 'collect') { // Ping-Pong Neon Synthesizer
        osc.type = 'square';
        osc.frequency.setValueAtTime(880, now); // A5
        osc.frequency.setValueAtTime(1760, now + 0.08); // Jump to A6 quickly
        
        gain.gain.setValueAtTime(0.15, now);
        gain.gain.exponentialRampToValueAtTime(0.01, now + 0.2);
        
        // Add a neon echo / delay effect
        const delay = audioCtx.createDelay();
        delay.delayTime.value = 0.12; 
        const feedback = audioCtx.createGain();
        feedback.gain.value = 0.3; // Echo intensity
        
        gain.connect(delay);
        delay.connect(feedback);
        feedback.connect(delay);
        
        gain.connect(audioCtx.destination);
        delay.connect(audioCtx.destination);
        
        osc.start(now);
        osc.stop(now + 0.2);
    } 
    else if (type === 'crash') {
        gain.connect(audioCtx.destination);
        osc.type = 'sawtooth';
        osc.frequency.setValueAtTime(150, now);
        osc.frequency.exponentialRampToValueAtTime(40, now + 0.3);
        gain.gain.setValueAtTime(0.4, now);
        gain.gain.exponentialRampToValueAtTime(0.01, now + 0.3);
        osc.start(now);
        osc.stop(now + 0.3);
    } 
    else if (type === 'click') {
        gain.connect(audioCtx.destination);
        osc.type = 'sine';
        osc.frequency.setValueAtTime(1200, now);
        gain.gain.setValueAtTime(0.1, now);
        gain.gain.exponentialRampToValueAtTime(0.01, now + 0.05);
        osc.start(now);
        osc.stop(now + 0.05);
    }
    else if (type === 'celebrate') { // 8-bit Win Fanfare
        gain.connect(audioCtx.destination);
        osc.type = 'square';
        
        // Fast major arpeggio (C5, E5, G5, C6, E6)
        osc.frequency.setValueAtTime(523.25, now);
        osc.frequency.setValueAtTime(659.25, now + 0.1);
        osc.frequency.setValueAtTime(783.99, now + 0.2);
        osc.frequency.setValueAtTime(1046.50, now + 0.3);
        osc.frequency.setValueAtTime(1318.51, now + 0.4);
        
        gain.gain.setValueAtTime(0.2, now);
        gain.gain.setValueAtTime(0.2, now + 0.4);
        gain.gain.exponentialRampToValueAtTime(0.01, now + 1.0);
        
        osc.start(now);
        osc.stop(now + 1.0);
    }
    else if (type === 'lose') { // Sad pitch drop
        gain.connect(audioCtx.destination);
        osc.type = 'sawtooth';
        
        osc.frequency.setValueAtTime(300, now);
        osc.frequency.exponentialRampToValueAtTime(40, now + 0.8);
        
        gain.gain.setValueAtTime(0.3, now);
        gain.gain.exponentialRampToValueAtTime(0.01, now + 0.8);
        
        osc.start(now);
        osc.stop(now + 0.8);
    }
}

function saveScore(name, finalScore, timeTaken) {
    let lb = JSON.parse(localStorage.getItem('cyberRunnerLB') || '[]');
    lb.push({ name: name, score: Math.floor(finalScore), time: timeTaken.toFixed(1) });
    // Sort descending by score. If scores are equal, sort by lowest time
    lb.sort((a, b) => {
        if (b.score !== a.score) return b.score - a.score;
        return a.time - b.time;
    });
    lb = lb.slice(0, 10); // keep only top 10
    localStorage.setItem('cyberRunnerLB', JSON.stringify(lb));
}

function createConfetti() {
  const canvas = document.getElementById('confetti');
  canvas.width = window.innerWidth;
  canvas.height = window.innerHeight;
  const ctx = canvas.getContext('2d');
  
  const confettiPieces = [];
  const colors = ['#ff00ff', '#00f0ff', '#ffff00', '#ff0066', '#00ff88'];
  
  for (let i = 0; i < 120; i++) {
    confettiPieces.push({
      x: Math.random() * canvas.width,
      y: Math.random() * canvas.height - canvas.height,
      vx: (Math.random() - 0.5) * 8,
      vy: Math.random() * 5 + 5,
      rotation: Math.random() * Math.PI * 2,
      vr: (Math.random() - 0.5) * 0.2,
      size: Math.random() * 5 + 4,
      color: colors[Math.floor(Math.random() * colors.length)],
      life: 1
    });
  }
  
  function animate() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    
    for (let piece of confettiPieces) {
      piece.y += piece.vy;
      piece.x += piece.vx;
      piece.rotation += piece.vr;
      piece.vy += 0.1;
      piece.life -= 0.008;
      
      ctx.save();
      ctx.globalAlpha = piece.life;
      ctx.translate(piece.x, piece.y);
      ctx.rotate(piece.rotation);
      ctx.shadowBlur = 15;
      ctx.shadowColor = piece.color;
      ctx.fillStyle = piece.color;
      ctx.fillRect(-piece.size/2, -piece.size/2, piece.size, piece.size);
      ctx.restore();
    }
    
    if (confettiPieces.some(p => p.life > 0)) {
      requestAnimationFrame(animate);
    }
  }
  animate();
}

function onKey(e) {
  if (!gameRunning) return;
  if (e.code === 'ArrowLeft' || e.code === 'KeyA') {
    if (playerLane > -1) playerLane--;
  }
  if (e.code === 'ArrowRight' || e.code === 'KeyD') {
    if (playerLane < 1) playerLane++;
  }
  if (e.code === 'Space' || e.code === 'ArrowUp' || e.code === 'KeyW') {
    if (!isJumping && !isSliding) {
      playNeonSound('jump');
      playerVY = JUMP_FORCE; // True Gravity implementation
      isJumping = true;
    }
  }
  if (e.code === 'ArrowDown' || e.code === 'KeyS') {
    if (!isJumping && !isSliding) {
      isSliding = true;
      slideTimer = 0.6; 
    } else if (isJumping) {
      playerVY = -JUMP_FORCE * 1.5; // Fast ground slam!
    }
  }
}

}

let spawnTimer = 0;
let buildingTimer = 0;
let runCycle = 0;

function update() {
  if (!gameRunning) return;
  
  const dtRaw = clock.getDelta();
  const dt = Math.min(dtRaw, 0.1); // Cap delta time to prevent physics explosion on lag
  totalTime += dt;
  
  // Base movement tied to time instead of frames
  const moveStep = gameSpeed * dt * 60; 

  // --- SMOOTH PLAYER MOVEMENT ---
  const targetX = lanePositions[playerLane + 1];
  player.position.x += (targetX - player.position.x) * (1 - Math.exp(-15 * dt)); // Buttery exponential lerp
  
  if (isJumping) {
    playerVY -= GRAVITY * dt; // Gravity curve
    playerY += playerVY * dt;
    if (playerY <= 0) {
      playerY = 0;
      playerVY = 0;
      isJumping = false;
    }
  }

  // Smooth Ducking scale/position logic
  if (isSliding) {
      slideTimer -= dt;
      if (slideTimer <= 0) isSliding = false;
  }
  const targetScaleY = isSliding ? 0.4 : 1.0;
  const targetPosY = isSliding && !isJumping ? -0.4 : (isJumping ? playerY : 0);
  
  player.scale.y += (targetScaleY - player.scale.y) * (1 - Math.exp(-20 * dt));
  if (!isJumping) {
      player.position.y += (targetPosY - player.position.y) * (1 - Math.exp(-20 * dt));
  } else {
      player.position.y = playerY;
  }
  
  runCycle += dt * 15;
  if (!isJumping && !isSliding) {
    player.getObjectByName('legL').rotation.x = Math.sin(runCycle) * 0.6;
    player.getObjectByName('legR').rotation.x = -Math.sin(runCycle) * 0.6;
    player.getObjectByName('shoeL').rotation.x = Math.sin(runCycle) * 0.6;
    player.getObjectByName('shoeR').rotation.x = -Math.sin(runCycle) * 0.6;
    player.getObjectByName('armL').rotation.x = -Math.sin(runCycle) * 0.5;
    player.getObjectByName('armR').rotation.x = Math.sin(runCycle) * 0.5;
  } else if (isSliding) {
    player.getObjectByName('armL').rotation.x = -Math.PI / 2;
    player.getObjectByName('armR').rotation.x = -Math.PI / 2;
  }

  // --- SMOOTH CAMERA SYSTEM ---
  const targetCamX = player.position.x * 0.3;
  const targetCamY = 4 + (player.position.y * 0.2);
  
  camera.position.x += (targetCamX - camera.position.x) * (1 - Math.exp(-10 * dt));
  camera.position.y += (targetCamY - camera.position.y) * (1 - Math.exp(-10 * dt));
  
  // Smooth Camera Tilt when changing lanes
  const targetTilt = -playerLane * 0.05;
  camera.rotation.z += (targetTilt - camera.rotation.z) * (1 - Math.exp(-10 * dt));
  
  // --- GUZAR SMOOTH CHASING ---
  guzarDistance += (targetGuzarDist - guzarDistance) * (1 - Math.exp(-2 * dt));
  guzar.position.z = player.position.z + guzarDistance;
  
  // Guzar smoothly follows player's lane X
  guzar.position.x += (player.position.x - guzar.position.x) * (1 - Math.exp(-4 * dt));
  
  // Animate Guzar
  guzar.getObjectByName('legL').rotation.x = Math.sin(runCycle * 1.1) * 0.7;
  guzar.getObjectByName('legR').rotation.x = -Math.sin(runCycle * 1.1) * 0.7;
  guzar.getObjectByName('bootL').rotation.x = Math.sin(runCycle * 1.1) * 0.7;
  guzar.getObjectByName('bootR').rotation.x = -Math.sin(runCycle * 1.1) * 0.7;
  guzar.getObjectByName('armL').rotation.x = -Math.sin(runCycle * 1.1) * 0.6;
  guzar.getObjectByName('armR').rotation.x = Math.sin(runCycle * 1.1) * 0.6;
  
  const arrow = guzar.getObjectByName('arrow');
  if (arrow) arrow.position.y = 2.25 + Math.sin(totalTime * 5) * 0.1;
  
  document.getElementById('guzarWarning').style.display = 'block';
  
  // --- WORLD UPDATES USING moveStep ---
  groundSegments.forEach(g => {
    g.position.z += moveStep;
    if (g.position.z > 30) g.position.z -= 120;
  });
  
  buildings.forEach(b => {
    b.position.z += moveStep;
  });
  buildings = buildings.filter(b => {
    if (b.position.z > 20) {
      scene.remove(b);
      return false;
    }
    return true;
  });
  
  buildingTimer += dt;
  if (buildingTimer > 0.4 / (gameSpeed / 0.4)) { // Scale spawn rate with speed
    buildingTimer = 0;
    spawnBuilding(-90);
  }
  
  if (!gateSpawned) {
    spawnTimer += dt;
    if (spawnTimer > 0.6 / (gameSpeed / 0.4)) {
      spawnTimer = 0;
      if (Math.random() < 0.6) spawnObstacle(-60);
      if (Math.random() < 0.7) spawnCard(-60);
    }
  }
  
  obstacles.forEach(o => o.position.z += moveStep);
  obstacles = obstacles.filter(o => {
    if (o.position.z > 10) {
      scene.remove(o);
      return false;
    }
    return true;
  });
  
  cards.forEach(c => {
    c.position.z += moveStep;
    c.rotation.y += 3 * dt;
    c.position.y = 1.2 + Math.sin(totalTime * 3 + c.position.z) * 0.15;
  });
  cards = cards.filter(c => {
    if (c.position.z > 10) {
      scene.remove(c);
      return false;
    }
    return true;
  });
  
  // Collisions
  cards.forEach((c, i) => {
    const dx = Math.abs(c.position.x - player.position.x);
    const dz = Math.abs(c.position.z - player.position.z);
    const dy = Math.abs(c.position.y - (player.position.y + 1));
    if (dx < 0.8 && dz < 0.8 && dy < 1.2) {
      scene.remove(c);
      cards.splice(i, 1);
      score += 37.5;
      playNeonSound('collect'); // Ping Pong sound effect
      updateHUD();
    }
  });
  
  obstacles.forEach(o => {
    const dx = Math.abs(o.position.x - player.position.x);
    const dz = Math.abs(o.position.z - player.position.z);
    if (dx < 0.7 && dz < 0.7) {
      if (o.userData.type === 'barricade' && playerY > 1.2) return;
      if (o.userData.type === 'papers' && playerY > 1.1) return;
      if (o.userData.type === 'guard' && playerY > 1.8) return;
      
      // Allow sliding under barricades (if Y is low enough)
      if (o.userData.type === 'barricade' && isSliding) return;
      
      let crashMsg = 'You crashed into ';
      if (o.userData.type === 'guard') crashMsg += 'a Security Guard!';
      else if (o.userData.type === 'barricade') crashMsg += 'a Barricade!';
      else crashMsg += 'a Pile of Papers!';
      
      playNeonSound('crash'); // Obstacle crash sound
      endGame(false, crashMsg);
    }
  });
  
  if (score >= TARGET_SCORE && !gateSpawned) {
    gateSpawned = true;
    obstacles.forEach(o => scene.remove(o));
    cards.forEach(c => scene.remove(c));
    obstacles = [];
    cards = [];
    spawnMainGate(-50);
  }
  
  if (mainGate) {
    mainGate.position.z += moveStep;
    
    const t = totalTime * 5;
    mainGate.userData.lights.forEach(l => {
      l.material.color.setHex(Math.sin(t) > 0 ? 0xff0000 : 0x440000);
    });
    
    if (!mainGate.userData.passed && mainGate.position.z > 1 && mainGate.position.z < 4) {
      mainGate.userData.passed = true;
      if (playerY > 1.6) {
        endGame(true, 'GATEPASS APPROVED!', false);
      } else {
        endGame(false, 'GATEPASS BLOCKED!', false);
      }
    }
  }
}

function updateHUD() {
  document.getElementById('score').textContent = Math.floor(score);
  const pct = Math.min(100, (score / TARGET_SCORE) * 100);
  document.getElementById('progressBar').style.width = pct + '%';
}

  gameEnded = true;
  gameRunning = false;
  document.getElementById('gateAlert').style.display = 'none';
  document.getElementById('guzarWarning').style.display = 'none';
  
  // Save Score locally 
  saveScore(playerData.name, score, totalTime);

  const cert = document.getElementById('certificate');
  if (success) {
    playNeonSound('celebrate');
    createConfetti();
    cert.className = 'certificate approved';
    cert.innerHTML = `

      <p style="font-size:24px;">CADET <span style="color:#ffff00">${playerData.name.toUpperCase()}</span></p>

      <hr style="margin:20px 0; opacity:0.5; border-color: currentColor;">
      <p>You made it through the gate!</p>
